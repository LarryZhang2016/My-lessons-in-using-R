3 Some useful tricks

3.1 Removing rows under constraints

We discuss two cases.

Case 1: removing rows that contain NA. For this case, we can use na.omit() or complete.cases(). Let us see the following example.

Example:

a_df <- 
  data_frame(x = c("a", NA, "b", "d", "e", "f"),
             y = c(runif(5), NA),
             z = c(LETTERS[1:3], NA, LETTERS[5:6]))
(a_df)
## # A tibble: 6 x 3
##   x           y z    
##   <chr>   <dbl> <chr>
## 1 a      0.138  A    
## 2 <NA>   0.233  B    
## 3 b      0.0726 C    
## 4 d      0.224  <NA> 
## 5 e      0.948  E    
## 6 f     NA      F
# using na.omit to remove rows that have NA
b_df <- 
  a_df %>% 
  na.omit()
(b_df)
## # A tibble: 3 x 3
##   x          y z    
##   <chr>  <dbl> <chr>
## 1 a     0.138  A    
## 2 b     0.0726 C    
## 3 e     0.948  E
# or using complete.cases
c_df <- 
  a_df[complete.cases(a_df) == TRUE, ] 
(c_df)
## # A tibble: 3 x 3
##   x          y z    
##   <chr>  <dbl> <chr>
## 1 a     0.138  A    
## 2 b     0.0726 C    
## 3 e     0.948  E

Case 2: removing rows contain some a specific symbol, e.g. “ZZZ”. Here the trick is to use rlang::sym.

Example:

library(dplyr)
a_df <- 
  data_frame(x = c("a", "ZZZ", "b", "d", "e", "f"),
             y = c(letters[1:5], "ZZZ"),
             z = c(LETTERS[1:3], "ZZZ", LETTERS[5:6]),
             w = rnorm(6))
(a_df)
## # A tibble: 6 x 4
##   x     y     z           w
##   <chr> <chr> <chr>   <dbl>
## 1 a     a     A      0.865 
## 2 ZZZ   b     B     -0.655 
## 3 b     c     C      0.0492
## 4 d     d     ZZZ   -0.424 
## 5 e     e     E     -1.06  
## 6 f     ZZZ   F      0.960
cols <- colnames(a_df)
b_df <- a_df
for(col in cols)
{b_df <-
  filter(b_df, (!!rlang::sym(col)) != "ZZZ")
}
(b_df)
## # A tibble: 3 x 4
##   x     y     z           w
##   <chr> <chr> <chr>   <dbl>
## 1 a     a     A      0.865 
## 2 b     c     C      0.0492
## 3 e     e     E     -1.06

NB: In the above, we must use parentheses around !!rlang::sym(col).

3.2 dplyr programming

The good references are:

  • https://cran.r-project.org/web/packages/dplyr/vignettes/programming.html

Here, we try several examples.

Example 1: square a variable in a dataframe, using rlang::sym() and !! (bang, bang)

library(dplyr)

sq_a_var <- function(df, var)
{new_var_name <- paste0(var, "_sq")

 new_df <-
  df %>%
  mutate(!!rlang::sym(new_var_name) := (!!rlang::sym(var))^2)

 new_df
}

a_df <- data.frame(x = 1:3,
                   y = 2:4)

(sq_a_var(a_df, "x"))
##   x y x_sq
## 1 1 2    1
## 2 2 3    4
## 3 3 4    9
(sq_a_var(a_df, "y"))
##   x y y_sq
## 1 1 2    4
## 2 2 3    9
## 3 3 4   16

Version 2, in which we use dplyr::enquo() and dplyr::quo_name() and !! (bang, bang).

library(dplyr)

sq_a_var <- function(df, var)
{var_enquo <- enquo(var)
  
 new_var_name <- paste0(quo_name(var_enquo), "_sq")

 new_df <-
  df %>%
  mutate(!!new_var_name := (!!var_enquo)^2)

 new_df
}

a_df <- data.frame(x = 1:3,
                   y = 2:4)

(sq_a_var(a_df, x))
##   x y x_sq
## 1 1 2    1
## 2 2 3    4
## 3 3 4    9
(sq_a_var(a_df, y))
##   x y y_sq
## 1 1 2    4
## 2 2 3    9
## 3 3 4   16

Notice that in the above we use bare x or bare y rather than “x” or “y”.

Example 2: find frequency of a variable, using enquo() and !! (bang, bang)

library(dplyr)

set.seed(20180425)
a_df <- data.frame(x = sample(letters[1:3], 6, replace = TRUE),
                   y = sample(LETTERS[1:3], 6, replace = TRUE))

freq_table <- function(df, var)
{n <- nrow(df)

 var_enqo <- enquo(var)
 
 re <- 
   df %>% 
   group_by(!!var_enqo) %>% 
   summarise(freq = n() / n)
 
 re
}

(freq_table(a_df, x))
## # A tibble: 3 x 2
##   x      freq
##   <fct> <dbl>
## 1 a     0.167
## 2 b     0.167
## 3 c     0.667
(freq_table(a_df, y))
## # A tibble: 3 x 2
##   y      freq
##   <fct> <dbl>
## 1 A     0.167
## 2 B     0.333
## 3 C     0.500

Example 3: group-by more than a variable, using quos() and !!!

library(dplyr)

set.seed(20180425)
a_df <- data.frame(x = sample(letters[1:3], 6, replace = TRUE),
                   y = sample(LETTERS[1:3], 6, replace = TRUE),
                   z = as.character(rep(1:3, each = 2)))

summ_a_few_vars <- function(df, ...)
{groupby_var <- quos(...)

 re <- 
   df %>% 
   group_by(!!!groupby_var) %>% 
   summarise(freq = n())
 
 re
}
(a_df)
##   x y z
## 1 b B 1
## 2 c B 1
## 3 c C 2
## 4 c C 2
## 5 a A 3
## 6 c C 3
(summ_a_few_vars(a_df, x, y))
## # A tibble: 4 x 3
## # Groups:   x [?]
##   x     y      freq
##   <fct> <fct> <int>
## 1 a     A         1
## 2 b     B         1
## 3 c     B         1
## 4 c     C         3
(summ_a_few_vars(a_df, y, z))
## # A tibble: 4 x 3
## # Groups:   y [?]
##   y     z      freq
##   <fct> <fct> <int>
## 1 A     3         1
## 2 B     1         2
## 3 C     2         2
## 4 C     3         1
